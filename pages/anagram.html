//anagram.java

import java.util.HashMap;

public class AnagramCheck {
    public static boolean areAnagrams(String str1, String str2) {
        // If lengths are not equal, they cannot be anagrams
        if (str1.length() != str2.length()) {
            return false;
        }

        // Count occurrences of each character in both strings
        HashMap countMap = new HashMap<>();

        for (char c : str1.toCharArray()) {
            countMap.put(c, countMap.getOrDefault(c, 0) + 1);
        }

        for (char c : str2.toCharArray()) {
            if (countMap.containsKey(c)) {
                countMap.put(c, countMap.get(c) - 1);
                if (countMap.get(c) < 0) {
                    return false;
                }
            } else {
                return false;
            }
        }

        return true; // All counts should be zero
    }

    public static void main(String[] args) {
        String str1 = "listen"; // Test case 1
        String str2 = "silent"; // Test case 1

        if (areAnagrams(str1, str2)) {
            System.out.println(str1 + " and " + str2 + " are anagrams.");
        } else {
            System.out.println(str1 + " and " + str2 + " are not anagrams.");
        }
    }
}

#include <stdio.h>
#include <string.h>
#include <stdbool.h>

#define ALPHABET_SIZE 26

bool are_anagrams(char* str1, char* str2) {
    int count[ALPHABET_SIZE] = {0}; // Array to count occurrences of each letter

    // If lengths are not equal, they cannot be anagrams
    if (strlen(str1) != strlen(str2)) {
        return false;
    }

    // Count occurrences of each character in both strings
    for (int i = 0; str1[i] && str2[i]; i++) {
        count[str1[i] - 'a']++;   // Increment count for str1
        count[str2[i] - 'a']--;   // Decrement count for str2
    }

    // If all counts are zero, then they are anagrams
    for (int i = 0; i < ALPHABET_SIZE; i++) {
        if (count[i] != 0) {
            return false;
        }
    }
    
    return true;
}

        printf("%s and %s are anagrams.\n", str1, str2);
    } else {
        printf("%s and %s are not anagrams.\n", str1, str2);
    }

    return 0;
}

def are_anagrams(str1, str2):
    # If lengths are not equal, they cannot be anagrams
    if len(str1) != len(str2):
        return False

    # Count occurrences of each character in both strings
    count = {}
    for char in str1:
        count[char] = count.get(char, 0) + 1

    for char in str2:
        if char in count:
            count[char] -= 1
        else:
            return False

    # If all counts are zero, then they are anagrams
    return all(value == 0 for value in count.values())
