//sub.java

public class MaxSubarraySum {

    public static int findMaxSubarraySum(int[] arr) {
        int maxCurrent = arr[0];  // Maximum sum including the current element
        int maxGlobal = arr[0];   // Maximum sum encountered so far

        for (int i = 1; i < arr.length; i++) {
            maxCurrent = Math.max(arr[i], maxCurrent + arr[i]);
            if (maxCurrent > maxGlobal) {
                maxGlobal = maxCurrent;
            }
        }
        return maxGlobal;
    }

    public static void main(String[] args) {
        int[] arr = {-2, 1, -3, 4, -1, 2, 1, -5, 4};
        int maxSum = findMaxSubarraySum(arr);
        System.out.println("Maximum subarray sum: " + maxSum);
    }
}
                
                
//sub.c

#include <stdio.h>
#include <limits.h>  // For INT_MIN

int findMaxSubarraySum(int arr[], int size) {
    int max_current = arr[0];  // Maximum sum including the current element
    int max_global = arr[0];   // Maximum sum encountered so far

    for (int i = 1; i < size; i++) {
        max_current = (arr[i] > (max_current + arr[i])) ? arr[i] : (max_current + arr[i]);
        if (max_current > max_global) {
            max_global = max_current;
        }
    }

    return max_global;
}

int main() {
    int arr[] = {-2, 1, -3, 4, -1, 2, 1, -5, 4};
    int size = sizeof(arr) / sizeof(arr[0]);

    int max_sum = findMaxSubarraySum(arr, size);
    printf("Maximum subarray sum: %d\n", max_sum);

    return 0;
}

def find_max_subarray_sum(arr):
    max_current = arr[0]  # Maximum sum including the current element
    max_global = arr[0]   # Maximum sum encountered so far

    for i in range(1, len(arr)):
        max_current = max(arr[i], max_current + arr[i])
        if max_current > max_global:
            max_global = max_current

    return max_global

# Example usage
arr = [-2, 1, -3, 4, -1, 2, 1, -5, 4]
max_sum = find_max_subarray_sum(arr)
print("Maximum subarray sum:", max_sum)
