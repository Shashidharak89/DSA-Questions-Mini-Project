Sum of 2 numbers

//sum.java

public class SumTwoDigits {
    public static void main(String[] args) {
        // Declare two variables for the digits
        int num1 = 5; // Example first digit
        int num2 = 7; // Example second digit

        // Calculate the sum
        int sum = num1 + num2;

        // Display the result
        System.out.println("The sum of " + num1 + " and " + num2 + " is: " + sum);
    }
}
                    
                
//sum.c

#include 
    int main() {
        // Two variables to hold the digits
        int num1 = 5;
        int num2 = 7;
    
        // Calculating the sum
        int sum = num1 + num2;
    
        // Output the result
        printf("The sum of %d and %d is: %d\n", num1, num2, sum);
    
        return 0;
    }
